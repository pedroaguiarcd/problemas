
nomedoaluno=input('nome do aluno: ')
notabimestral=int(input('digite a primeira nota do aluno: '))
notamensal=int(input('digite a segunda nota do aluno: '))
notatotal=notabimestral+notamensal
mediadoaluno=notatotal**1/2

print(f'A soma das notas do aluno {nomedoaluno} é {notabimestral+notamensal}\n logo a media do aluno é {mediadoaluno}')
